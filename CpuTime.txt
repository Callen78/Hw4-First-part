// my class that handles the CPU time of the computer package goldenwarriorsstate;/** * * @author CEOCarlAllen */public class TheShotClock {    private final long start;        public TheShotClock() {        start = System.currentTimeMillis();    }    public double elapsedTime() {        long now = System.currentTimeMillis();        return (now - start) / 1000.0;    }}
